<script>
document.addEventListener("DOMContentLoaded", function () {
  const toggles = document.querySelectorAll(".accordion-toggle");

  toggles.forEach(toggle => {
    toggle.addEventListener("click", function () {
      const container = this.closest(".accordion-section, .accordion-subsection");
      const content = this.nextElementSibling;
      const isOpen = container.classList.contains("open");

      container.classList.toggle("open", !isOpen);
      this.setAttribute("aria-expanded", String(!isOpen));

      if (content) {
        content.hidden = isOpen;
      }
    });
  });

  // Default state setup
  document.querySelectorAll(".accordion-section").forEach(section => {
    const content = section.querySelector(":scope > .accordion-content");
    const isOpen = section.classList.contains("open");
    if (content) content.hidden = !isOpen;

    const toggle = section.querySelector(":scope > .accordion-toggle");
    if (toggle) toggle.setAttribute("aria-expanded", String(isOpen));
  });

  document.querySelectorAll(".accordion-subsection").forEach(section => {
    const content = section.querySelector(":scope > .accordion-content");
    const isOpen = section.classList.contains("open");
    if (content) content.hidden = !isOpen;

    const toggle = section.querySelector(":scope > .accordion-toggle");
    if (toggle) toggle.setAttribute("aria-expanded", String(isOpen));
  });
});
</script>
